// --- 1. CONFIGURATION --- // Replace these with your actual Supabase URL and Key const _URL = "YOUR_SUPABASE_URL"; const _KEY = "YOUR_SUPABASE_ANON_KEY"; const supabase = supabase.createClient(_URL, _KEY); // --- 2. FETCH AND DISPLAY RECORDS --- async function fetchRecords() { console.log("DEBUG: Fetching records..."); const { data, error } = await supabase .from('students') .select('*') .order('created_at', { ascending: false }); if (error) { alert("FETCH ERROR: " + error.message); return; } const tableBody = document.getElementById('recordTableBody'); if (!tableBody) return; tableBody.innerHTML = ""; if (data.length === 0) { tableBody.innerHTML = "No records found."; return; } data.forEach((student, index) => { // We calculate age here based on the DOB in the database const birthDate = new Date(student.dob); const age = student.dob ? (new Date().getFullYear() - birthDate.getFullYear()) : "N/A"; const row = ` ${index + 1} ${student.gen_lin || 'N/A'} ${student.official_lin || 'N/A'} ${student.name || 'No Name'} ${age} ${student.location || 'N/A'} ${student.student_status || 'Active'} `; tableBody.insertAdjacentHTML('beforeend', row); }); } // --- 3. UPLOAD RECORD --- async function registerStudent() { const btn = document.querySelector("button"); const originalText = btn.innerText; // Change button state btn.innerText = "UPLOADING..."; btn.disabled = true; // Get values from your form const nameValue = document.getElementById('regName').value; const dobValue = document.getElementById('regDOB').value; const locationValue = document.getElementById('regLocation').value; const healthValue = document.getElementById('regHealth').value; const officialLINValue = document.getElementById('regOfficialLIN').value; const contactValue = document.getElementById('regContact').value; // Generate a unique LIN for the record const generatedLIN = `PM${nameValue.charAt(0).toUpperCase()}${Math.floor(1000 + Math.random() * 9000)}`; const studentData = { name: nameValue, dob: dobValue, location: locationValue, // Matches your 'location' column in Supabase health_status: healthValue, official_lin: officialLINValue, contact_number: contactValue, gen_lin: generatedLIN, student_status: 'Active' }; const { error } = await supabase .from('students') .insert([studentData]); if (error) { alert("UPLOAD FAILED: " + error.message); btn.innerText = originalText; btn.disabled = false; } else { alert("SUCCESS: Record saved!"); // Reset form and button btn.innerText = originalText; btn.disabled = false; // Refresh the table to show the new record fetchRecords(); } } // Initialize the app window.onload = fetchRecords;
